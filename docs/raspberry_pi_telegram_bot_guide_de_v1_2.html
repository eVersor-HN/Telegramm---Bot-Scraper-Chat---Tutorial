Allgemeiner Bot-Code

Generischer Code: normale Nachrichten werden sofort gespeichert.

bot.py schreiben
cd ~/telegram-bot
cat > bot.py <<'BOTPY'
import json
import os
import re
import sqlite3
from datetime import datetime
from pathlib import Path
from zoneinfo import ZoneInfo

from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, ContextTypes, filters

BASE = Path(__file__).resolve().parent
DB_FILE = BASE / "bot.db"
ENV_FILE = BASE / ".env"
BACKUP_DIR = BASE / "backups"
TZ = ZoneInfo("Europe/Berlin")

def load_env():
    if not ENV_FILE.exists():
        return
    for line in ENV_FILE.read_text(encoding="utf-8", errors="ignore").splitlines():
        line = line.strip()
        if not line or line.startswith("#") or "=" not in line:
            continue
        key, value = line.split("=", 1)
        os.environ.setdefault(key.strip(), value.strip().strip('"').strip("'"))

def now_local():
    return datetime.now(TZ)

def clean_text(text):
    text = str(text or "").strip()
    text = re.sub(r"\s+", " ", text)
    for before, after in {" ,": ",", " .": ".", " !": "!", " ?": "?"}.items():
        text = text.replace(before, after)
    return text.strip()

def db():
    con = sqlite3.connect(DB_FILE)
    con.row_factory = sqlite3.Row
    return con

def init_db():
    con = db()
    cur = con.cursor()
    cur.execute("""
        CREATE TABLE IF NOT EXISTS notes (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            created_at TEXT NOT NULL,
            text TEXT NOT NULL
        )
    """)
    con.commit()
    con.close()

def add_note(text):
    text = clean_text(text)
    if not text:
        raise ValueError("empty note")
    con = db()
    cur = con.cursor()
    cur.execute("INSERT INTO notes (created_at, text) VALUES (?, ?)", (now_local().isoformat(timespec="seconds"), text))
    note_id = cur.lastrowid
    con.commit()
    con.close()
    return note_id

def get_last(limit=10):
    con = db()
    rows = con.execute("SELECT id, created_at, text FROM notes ORDER BY id DESC LIMIT ?", (limit,)).fetchall()
    con.close()
    return list(reversed(rows))

def format_notes(rows):
    if not rows:
        return "VAULT.EMPTY // no notes stored"
    return "\n".join(f"- #{r['id']} {r['text']}" for r in rows)

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "[RASPI.BOT]\nNODE ONLINE // send normal text to store a note\n\n"
        "/status - health check\n/last - latest notes\n/files - export backups\n/help - command list"
    )

async def status(update: Update, context: ContextTypes.DEFAULT_TYPE):
    con = db()
    total = con.execute("SELECT COUNT(*) AS c FROM notes").fetchone()["c"]
    con.close()
    await update.message.reply_text(f"STATUS // ONLINE\nSTORED // {total} notes")

async def last(update: Update, context: ContextTypes.DEFAULT_TYPE):
    limit = 10
    if context.args:
        try:
            limit = max(1, min(50, int(context.args[0])))
        except ValueError:
            limit = 10
    await update.message.reply_text("[LATEST]\n" + format_notes(get_last(limit)))

async def files(update: Update, context: ContextTypes.DEFAULT_TYPE):
    BACKUP_DIR.mkdir(exist_ok=True)
    stamp = now_local().strftime("%Y%m%d-%H%M%S")
    txt = BACKUP_DIR / f"notes-{stamp}.txt"
    jsonl = BACKUP_DIR / f"notes-{stamp}.jsonl"
    rows = get_last(100000)
    txt.write_text(format_notes(rows) + "\n", encoding="utf-8")
    with jsonl.open("w", encoding="utf-8") as f:
        for r in rows:
            f.write(json.dumps(dict(r), ensure_ascii=False) + "\n")
    await update.message.reply_text("FILE.NODE // exporting backup files")
    await update.message.reply_document(document=txt.open("rb"), filename=txt.name)
    await update.message.reply_document(document=jsonl.open("rb"), filename=jsonl.name)

async def help_cmd(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await start(update, context)

async def store_text(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = clean_text(update.message.text)
    if not text:
        await update.message.reply_text("INTAKE.REJECTED // empty message")
        return
    note_id = add_note(text)
    await update.message.reply_text(f"[LOGGED]\n- #{note_id} {text}")

def main():
    load_env()
    token = os.environ.get("BOT_TOKEN", "").strip()
    if not token:
        raise RuntimeError("BOT_TOKEN missing in .env")
    init_db()
    app = Application.builder().token(token).build()
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("help", help_cmd))
    app.add_handler(CommandHandler("status", status))
    app.add_handler(CommandHandler("last", last))
    app.add_handler(CommandHandler("files", files))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, store_text))
    app.run_polling(allowed_updates=Update.ALL_TYPES, drop_pending_updates=True)

if __name__ == "__main__":
    main()

Autostart nach Neustart und Stromausfall

Als Service installieren

systemd startet den Bot beim Booten und nach Absturz neu.

Service erstellen und aktivieren
sudo tee /etc/systemd/system/telegram-bot.service > /dev/null <<'SERVICE'
[Unit]
Description=Raspberry Pi Telegram Bot
Wants=network-online.target
After=network-online.target

[Service]
Type=simple
User=pi
WorkingDirectory=/home/pi/telegram-bot
EnvironmentFile=/home/pi/telegram-bot/.env
ExecStart=/home/pi/telegram-bot/venv/bin/python /home/pi/telegram-bot/bot.py
Restart=always
RestartSec=10
TimeoutStartSec=60

[Install]
WantedBy=multi-user.target
SERVICE

sudo systemctl daemon-reload
sudo systemctl enable telegram-bot
sudo systemctl start telegram-bot
systemctl status telegram-bot --no-pager -l
Wenn dein User nicht pi heisst, User=pi und /home/pi ersetzen.

FAQ

Brauche ich einen Raspberry Pi 5?
Nein. Fuer einen einfachen Telegram-Bot reichen Pi 3 oder Pi 4. Pi 5 lohnt sich fuer mehr Automatisierung oder lokale KI-Tests.
Startet der Bot nach Stromausfall wieder?
Ja, wenn der systemd-Service aktiviert ist und der Pi normal bootet.
Warum systemd?
systemd startet den Bot beim Booten, startet ihn nach Abstuerzen neu und liefert Logs ueber journalctl.
Was ist BotFather?
BotFather ist der offizielle Telegram-Bot zum Erstellen und Verwalten eigener Bots.
Wo kommt der Token hin?
In die Datei .env, nie direkt in bot.py und nie in GitHub.
Kann ich PuTTY nutzen?
Ja. Fuer Windows ist PuTTY sehr praktisch. Setze Close window on exit auf Never.
Warum schliesst sich PuTTY?
Meist beendet sich die Shell. Aendere die PuTTY-Einstellung oder nutze Bloecke ohne exit.
Kann ich die Bloecke direkt kopieren?
Ja. Vorher Platzhalter wie BOT_TOKEN und Username ersetzen.
Kann ich ohne nano bearbeiten?
Ja. Nutze komplette heredoc-Dateibloecke oder kleine Python-Patches.
Was wenn der Bot nicht antwortet?
Pruefe systemctl status und journalctl, dann Service neu starten.
Was wenn /today oder /last stark verzoegert ist?
Keine langsame KI im normalen Message-Handler nutzen. Erst speichern, spaeter auswerten.
Kann lokale KI jede Notiz klassifizieren?
Technisch ja, praktisch auf dem Pi oft zu langsam. Besser per explizitem Befehl.
Welche Datenbank wird genutzt?
SQLite. Einfach, lokal, ohne extra Server.
Kann ich den Bot oeffentlich betreiben?
Der Bot ist ueber Telegram erreichbar. SSH nicht offen ins Internet legen, wenn du es nicht absichern kannst.
Kann ich mehrere Bots betreiben?
Ja, aber mit getrennten Ordnern, Services und BotFather-Tokens.
Was passiert bei zwei Prozessen mit gleichem Token?
Es kann Polling-Konflikte geben. Doppelte Services stoppen.
Wie finde ich laufende Python-Bots?
Mit ps -ef | grep -i python und systemctl list-units.
Wie update ich Code sicher?
Service stoppen, Datei sichern, Code ersetzen, py_compile, Service starten.
Was ist drop_pending_updates?
Alte Telegram-Updates werden beim Start verworfen.
Kann der Bot Dateien senden?
Ja. Das Beispiel /files exportiert Backups als TXT und JSONL.
Kann ich Backups auf dem Pi behalten?
Ja, aber zusaetzlich extern kopieren. SD-Karten fallen aus.
Soll ich eine SSD nutzen?
Fuer dauerhaften Betrieb ist SSD robuster als billige microSD.
Ist WLAN okay?
Ja, fuer viele Bots. Ethernet ist stabiler.
Garantiert network-online.target dauerhaft Internet?
Nein. Es wartet nur beim Start. Spaetere Netzabbrueche muss der Bot vertragen.
Was wenn Telegram nicht erreichbar ist?
Der Service startet neu und sollte sich erholen, sobald das Netz wieder da ist.
Welcher Pi ist fuer lokale KI am besten?
Pi 5 mit 8GB oder 16GB. Trotzdem bleibt es CPU-limitiert.
Welche Modelle sind realistisch?
0.5B bis 3B quantisiert sind praktisch. 4B geht eher auf Pi 5 16GB. 7B ist nicht fluessig.
Kann ich Huihui-Modelle nutzen?
Ja, aber aktuelle Tags pruefen und klein testen. Abliterated-Modelle privat und beaufsichtigt nutzen.
Brauche ich Docker?
Nein. Diese Anleitung nutzt Python direkt und systemd.
Geht Raspberry Pi OS Desktop?
Ja, aber Lite braucht weniger Ressourcen.
Wie aendere ich den Bot-Namen?
In BotFather aendern und Text in bot.py anpassen.
Wie fuege ich Befehle hinzu?
Async-Funktion schreiben und mit CommandHandler registrieren.
Warum py_compile?
Es findet Syntaxfehler vor dem Neustart.
Welche Lizenz passt?
Wenn jeder alles frei nutzen darf, ist CC0 am einfachsten.
Kann ich das Repo veroeffentlichen?
Ja, solange keine Tokens, Datenbanken, Chat-IDs oder privaten Logs enthalten sind.
Was teste ich am Ende?
Manueller Start, Telegram-Antwort, systemd-Start, Reboot, Logs und Backup-Export.
