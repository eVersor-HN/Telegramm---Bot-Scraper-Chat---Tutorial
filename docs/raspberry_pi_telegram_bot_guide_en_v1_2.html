Instant notes, status and file export

Generic bot code

Generic code: stores text immediately and does not call AI during normal intake.

Write bot.py
cd ~/telegram-bot
cat > bot.py <<'BOTPY'
import json
import os
import re
import sqlite3
from datetime import datetime
from pathlib import Path
from zoneinfo import ZoneInfo

from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, ContextTypes, filters

BASE = Path(__file__).resolve().parent
DB_FILE = BASE / "bot.db"
ENV_FILE = BASE / ".env"
BACKUP_DIR = BASE / "backups"
TZ = ZoneInfo("Europe/Berlin")

def load_env():
    if not ENV_FILE.exists():
        return
    for line in ENV_FILE.read_text(encoding="utf-8", errors="ignore").splitlines():
        line = line.strip()
        if not line or line.startswith("#") or "=" not in line:
            continue
        key, value = line.split("=", 1)
        os.environ.setdefault(key.strip(), value.strip().strip('"').strip("'"))

def now_local():
    return datetime.now(TZ)

def clean_text(text):
    text = str(text or "").strip()
    text = re.sub(r"\s+", " ", text)
    for before, after in {" ,": ",", " .": ".", " !": "!", " ?": "?"}.items():
        text = text.replace(before, after)
    return text.strip()

def db():
    con = sqlite3.connect(DB_FILE)
    con.row_factory = sqlite3.Row
    return con

def init_db():
    con = db()
    cur = con.cursor()
    cur.execute("""
        CREATE TABLE IF NOT EXISTS notes (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            created_at TEXT NOT NULL,
            text TEXT NOT NULL
        )
    """)
    con.commit()
    con.close()

def add_note(text):
    text = clean_text(text)
    if not text:
        raise ValueError("empty note")
    con = db()
    cur = con.cursor()
    cur.execute("INSERT INTO notes (created_at, text) VALUES (?, ?)", (now_local().isoformat(timespec="seconds"), text))
    note_id = cur.lastrowid
    con.commit()
    con.close()
    return note_id

def get_last(limit=10):
    con = db()
    rows = con.execute("SELECT id, created_at, text FROM notes ORDER BY id DESC LIMIT ?", (limit,)).fetchall()
    con.close()
    return list(reversed(rows))

def format_notes(rows):
    if not rows:
        return "VAULT.EMPTY // no notes stored"
    return "\n".join(f"- #{r['id']} {r['text']}" for r in rows)

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "[RASPI.BOT]\nNODE ONLINE // send normal text to store a note\n\n"
        "/status - health check\n/last - latest notes\n/files - export backups\n/help - command list"
    )

async def status(update: Update, context: ContextTypes.DEFAULT_TYPE):
    con = db()
    total = con.execute("SELECT COUNT(*) AS c FROM notes").fetchone()["c"]
    con.close()
    await update.message.reply_text(f"STATUS // ONLINE\nSTORED // {total} notes")

async def last(update: Update, context: ContextTypes.DEFAULT_TYPE):
    limit = 10
    if context.args:
        try:
            limit = max(1, min(50, int(context.args[0])))
        except ValueError:
            limit = 10
    await update.message.reply_text("[LATEST]\n" + format_notes(get_last(limit)))

async def files(update: Update, context: ContextTypes.DEFAULT_TYPE):
    BACKUP_DIR.mkdir(exist_ok=True)
    stamp = now_local().strftime("%Y%m%d-%H%M%S")
    txt = BACKUP_DIR / f"notes-{stamp}.txt"
    jsonl = BACKUP_DIR / f"notes-{stamp}.jsonl"
    rows = get_last(100000)
    txt.write_text(format_notes(rows) + "\n", encoding="utf-8")
    with jsonl.open("w", encoding="utf-8") as f:
        for r in rows:
            f.write(json.dumps(dict(r), ensure_ascii=False) + "\n")
    await update.message.reply_text("FILE.NODE // exporting backup files")
    await update.message.reply_document(document=txt.open("rb"), filename=txt.name)
    await update.message.reply_document(document=jsonl.open("rb"), filename=jsonl.name)

async def help_cmd(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await start(update, context)

async def store_text(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = clean_text(update.message.text)
    if not text:
        await update.message.reply_text("INTAKE.REJECTED // empty message")
        return
    note_id = add_note(text)
    await update.message.reply_text(f"[LOGGED]\n- #{note_id} {text}")

def main():
    load_env()
    token = os.environ.get("BOT_TOKEN", "").strip()
    if not token:
        raise RuntimeError("BOT_TOKEN missing in .env")
    init_db()
    app = Application.builder().token(token).build()
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("help", help_cmd))
    app.add_handler(CommandHandler("status", status))
    app.add_handler(CommandHandler("last", last))
    app.add_handler(CommandHandler("files", files))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, store_text))
    app.run_polling(allowed_updates=Update.ALL_TYPES, drop_pending_updates=True)

if __name__ == "__main__":
    main()

Autostart after reboot and power loss

Install as service

systemd starts the bot at boot and restarts it after crashes.

Create and enable service
sudo tee /etc/systemd/system/telegram-bot.service > /dev/null <<'SERVICE'
[Unit]
Description=Raspberry Pi Telegram Bot
Wants=network-online.target
After=network-online.target

[Service]
Type=simple
User=pi
WorkingDirectory=/home/pi/telegram-bot
EnvironmentFile=/home/pi/telegram-bot/.env
ExecStart=/home/pi/telegram-bot/venv/bin/python /home/pi/telegram-bot/bot.py
Restart=always
RestartSec=10
TimeoutStartSec=60

[Install]
WantedBy=multi-user.target
SERVICE

sudo systemctl daemon-reload
sudo systemctl enable telegram-bot
sudo systemctl start telegram-bot
systemctl status telegram-bot --no-pager -l
If your username is not pi, replace User=pi and /home/pi.

FAQ

Do I need a Raspberry Pi 5?
No. A Pi 3 or Pi 4 is enough for a simple Telegram bot. Use Pi 5 mainly for heavier automation or local AI tests.
Will the bot restart after a power cut?
Yes, if the systemd service is enabled with systemctl enable and the Pi boots normally.
Why use systemd?
It starts the bot at boot, restarts it after crashes and gives you journalctl logs.
What is BotFather?
BotFather is Telegram's official bot used to create and manage bot accounts.
Where do I put the token?
Put it in .env, never directly in bot.py and never in GitHub.
Can I use PuTTY?
Yes. PuTTY is ideal on Windows. Set Close window on exit to Never so errors stay visible.
Why does PuTTY close after commands?
Usually because the shell exits. Change PuTTY behavior or avoid command blocks with exit statements.
Can I copy/paste the blocks directly?
Yes. Replace placeholders like BOT_TOKEN and username first.
Can I edit the bot without nano?
Yes. Use heredoc replacement blocks or Python patch snippets.
What if the bot stops replying?
Check systemctl status and journalctl, then restart the service.
What if /today or /last is delayed?
Avoid slow AI in the normal message handler. Store first, process later.
Can local AI classify every note?
Technically yes, but on Raspberry Pi it can make the bot feel broken. Use explicit commands.
Which database is used?
SQLite. It is simple, local and needs no server.
Can I expose the bot publicly?
The Telegram bot is reachable through Telegram. Do not expose SSH to the open internet unless you know what you are doing.
Can I run multiple bots?
Yes, but use separate directories, services and BotFather tokens.
What if two processes use the same token?
Telegram polling conflicts can happen. Stop duplicate services.
How do I see running Python bots?
Use ps -ef | grep -i python and systemctl list-units.
How do I update code safely?
Stop service, backup file, replace code, py_compile, restart.
What is drop_pending_updates?
It discards old queued Telegram updates when the bot starts.
Can the bot send files?
Yes. The example /files command exports backups as TXT and JSONL.
Can I keep backups on the Pi?
Yes, but copy them off-device too. SD cards fail.
Should I use an SSD?
For serious always-on use, SSD is more robust than cheap microSD.
Is Wi-Fi okay?
Yes for most bots. Ethernet is better for reliability.
Does network-online.target guarantee internet forever?
No. It only delays startup until network is considered online. The bot must handle later drops.
What if DNS or Telegram is unreachable?
The service restarts and the bot should recover when network returns.
Which Pi is best for local AI?
Pi 5 8GB or 16GB. Still expect CPU-bound speeds.
Which models are realistic?
0.5B to 3B quantized models are the practical range. 4B can be usable on Pi 5 16GB. 7B is not fluid.
Can I use Huihui models?
Yes, but check current tags and run small tests. Abliterated models should be private and supervised.
Do I need Docker?
No. This guide uses native Python and systemd.
Can I use a desktop Raspberry Pi OS?
Yes, but Lite wastes fewer resources.
How do I change the bot name?
Change it in BotFather and adjust text in bot.py if needed.
How do I add a command?
Write an async function and register it with CommandHandler.
Why do I run py_compile?
It catches syntax errors before restarting the service.
What license should I use?
If everyone may reuse everything, CC0 is the simplest public-domain-style choice.
Can I publish this repo?
Yes, if it contains only examples and no tokens, databases, chat IDs or personal logs.
What should I test before calling it finished?
Manual run, Telegram reply, systemd start, reboot recovery, logs and backup export.
